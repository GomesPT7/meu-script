local TP = MakeTab({Name = "Teleport"})

-- Variables for auto teleport
local autoTeleportEnabled = false
local teleportThread

-- Function to start auto teleport
local function startAutoTeleport()
    autoTeleportEnabled = true
    teleportThread = coroutine.create(function()
        while autoTeleportEnabled do
            local nearPosition = CFrame.new(-100, 5, -1600)  -- Adjust position as needed
            player.Character.HumanoidRootPart.CFrame = nearPosition
            print("Teleported to a nearby position:", nearPosition.Position)
            
            wait(3)
            
            local targetBlock = workspace.Parkour.Wins:FindFirstChild("14")
            
            if targetBlock then
                player.Character.HumanoidRootPart.CFrame = targetBlock.CFrame
                print("Teleported to:", targetBlock.Position)

                wait(3)
                wait(2)
            else
                print("Target block not found.")
                wait(2)
            end
        end
    end)
    coroutine.resume(teleportThread)
    MakeNotifi({
        Title = "Auto Teleport",
        Text = "Auto teleport activated!",
        Time = 3
    })
end

-- Function to stop auto teleport
local function stopAutoTeleport()
    autoTeleportEnabled = false
    if teleportThread then
        coroutine.close(teleportThread)
        teleportThread = nil
    end
    MakeNotifi({
        Title = "Auto Teleport",
        Text = "Auto teleport deactivated!",
        Time = 3
    })
end

-- Button to toggle auto teleport
AddButton(TP, {
    Name = "Toggle Auto Farm +14",
    Callback = function()
        if autoTeleportEnabled then
            stopAutoTeleport()
            print("Auto teleport deactivated.")
        else
            startAutoTeleport()
            print("Auto teleport activated.")
        end
    end
})
